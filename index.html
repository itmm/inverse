template<int N>
void identity(int (&p)[N]) {
for (
int i { 1 }; i <= N; ++i
) {
p[i - 1] = i;
}
}
@end(globals)
@add(globals)
template<int N>
bool is_identity(
const int (&p)[N]
) {
for (
int i { 1 }; i <= N; ++i
) {
if (p[i - 1] != i) {
return false;
}
}
return true;
}

@add(globals)
template <int N>
void mult(
int (&p)[N],
const int (&o)[N]
) {
assert(&p != &o);
for (
int i { 1 }; i <= N; ++i
) {
p[i - 1] =
o[p[i - 1] - 1];
}
}
@end(globals)

void inv(int (&p)[N]) {
@put(inv);

@end(globals)
@def(inv)
int m = N, j = -1;
while (m > 0) {
int i = p[m - 1];
while (i > 0) {
p[m - 1] = j; j = -m;
m = i; i = p[m - 1];
if (i < 0) {
i = j; break;
}
}
p[m - 1] = -i;
--m;
}
@end(inv)
